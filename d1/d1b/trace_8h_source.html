44 inline QCString trunc(const QCString &s,size_t numChars=15)
45 {
46 QCString result;
47 size_t i=0;
48 for (; i<numChars && i<s.length(); i++)
49 {
50 char c=s.at(i);
51 if (c=='\n') { result+="\\n"; }
52 else if (c=='\t') { result+="\\t"; }
53 else if (c=='\r') { result+="\\r"; }
54 else if (c=='\\') { result+="\\\\"; }
55 else result+=c;
56 }
57 if (i<s.length()) result+="...";
58 return result;
59 }
60}

68/** Helper class to trace an entry statement at creation and another one at destruction. */
69class AutoTrace
70{
71 public:
72 explicit AutoTrace(spdlog::source_loc loc) : m_loc(loc)
73 {
74 if (g_tracer)
75 {
76 g_tracer->log(m_loc,spdlog::level::trace,">");
77 }
78 }
79 template<typename... Args>
80 explicit AutoTrace(spdlog::source_loc loc,
81 const std::string &fmt, Args&&...args) : m_loc(loc)
82 {
83 if (g_tracer)
84 {
85 if (fmt.empty())
86 {
87 g_tracer->log(m_loc,spdlog::level::trace,">");
88 }
89 else
90 {
91 g_tracer->log(m_loc,spdlog::level::trace,"> "+fmt,std::forward<Args>(args)...);
92 }
93 }
94 }
95 ~AutoTrace()
96 {
97 if (g_tracer)
98 {
99 if (m_exitMessage.empty())
100 {
101 g_tracer->log(m_loc,spdlog::level::trace,"<");
102 }
103 else
104 {
105 g_tracer->log(m_loc,spdlog::level::trace,"< "+m_exitMessage);
106 }
107 }
108 }
109 template<typename... Args>
110 void add(spdlog::source_loc loc,
111 const std::string &fmt, Args&&...args)
112 {
113 if (g_tracer)
114 {
115 g_tracer->log(loc,spdlog::level::trace,": "+fmt,std::forward<Args>(args)...);
116 }
117 }
118 template<typename... Args>
119 void setExit(const std::string &msg,Args&&...args)
120 {
121 m_exitMessage = fmt::format(msg,std::forward<Args>(args)...);
122 }
123 private:
124 spdlog::source_loc m_loc;
125 std::string m_exitMessage;
126};
