79 * spaces visible, should break lines at a newline and should convert

84 /*! Writes a link to an object in a code fragment.
85 * \param type The type of symbol, used for semantic syntax
86 * highlighting, may be Default is no info is available.
87 * \param ref If this is non-zero, the object is to be found in
88 * an external documentation file.
89 * \param file The file in which the object is located.
90 * \param anchor The anchor uniquely identifying the object within
91 * the file.
92 * \param name The text to display as a placeholder for the link.
93 * \param tooltip The tooltip to display when the mouse is on the link.
94 */
95 virtual void writeCodeLink(CodeSymbolType type,
96 const QCString &ref,const QCString &file,
97 const QCString &anchor,const QCString &name,
98 const QCString &tooltip) = 0;
99
100 /*! Writes the line number of a source listing
101 * \param ref External reference (when imported from a tag file)
102 * \param file The file part of the URL pointing to the docs.
103 * \param anchor The anchor part of the URL pointing to the docs.
104 * \param lineNumber The line number to write
105 * \param writeLineAnchor Indicates if an anchor for the line number needs to be written
106 */
107 virtual void writeLineNumber(const QCString &ref,const QCString &file,
108 const QCString &anchor,int lineNumber, bool writeLineAnchor) = 0;
109
110 /*! Writes a tool tip definition
111 * \param id unique identifier for the tooltip
112 * \param docInfo Info about the symbol's documentation.
113 * \param decl full declaration of the symbol (for functions)
114 * \param desc brief description for the symbol
115 * \param defInfo Info about the symbol's definition in the source code
116 * \param declInfo Info about the symbol's declaration in the source code
117 */
118 virtual void writeTooltip(const QCString &id,
119 const DocLinkInfo &docInfo,
120 const QCString &decl,
121 const QCString &desc,
122 const SourceLinkInfo &defInfo,
123 const SourceLinkInfo &declInfo
124 ) = 0;
125
126 virtual void startCodeLine(bool hasLineNumbers) = 0;
127
128 /*! Ends a line of code started with startCodeLine() */
129 virtual void endCodeLine() = 0;
130
131 /*! Starts a block with a certain meaning. Used for syntax highlighting,
132 * which elements of the same type are rendered using the same 'font class'.
133 * \param clsName The category name.
134 */
135 virtual void startFontClass(const QCString &clsName) = 0;
136
137 /*! Ends a block started with startFontClass() */
138 virtual void endFontClass() = 0;
139
140 /*! Write an anchor to a source listing.
141 * \param name The name of the anchor.
142 */
143 virtual void writeCodeAnchor(const QCString &name) = 0;
144
145 /*! Starts a source code fragment. The fragment will be
146 * fed to the code parser (see code.h) for syntax highlighting
147 * and cross-referencing. The fragment ends by a call to
148 * endCodeFragment()
149 * @param style The kind of code fragment.
150 */
151 virtual void startCodeFragment(const QCString &style) = 0;
152 /*! Ends a block of code */
153 virtual void endCodeFragment(const QCString &style) = 0;
154
155 private:
156 int m_id = -1;
157 bool m_active = true;
