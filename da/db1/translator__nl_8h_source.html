346 QCString trWriteList(int numEntries)
347 {
348 // this function is used to produce a comma-separated list of items.
349 // use generateMarker(i) to indicate where item i should be put.
350 QCString result;
351 int i;
352 // the inherits list contain `numEntries' classes
353 for (i=0;i<numEntries;i++)
354 {
355 // use generateMarker to generate placeholders for the class links!
356 result+=generateMarker(i); // generate marker for entry i in the list
357 // (order is left to right)
358
359 if (i!=numEntries-1) // not the last entry, so we need a separator
360 {
361 if (i<numEntries-2) // not the fore last entry
362 result+=", ";
363 else // the fore last entry
364 result+=" en ";
365 }
366 }
367 return result;
368 }
