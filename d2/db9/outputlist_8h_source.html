144/** Class representing a list of different code generators.
145 * It offers the same interface as the specific code generators,
146 * and will forward each method to all generators that are enabled.
147 */
148class OutputCodeList
149{
150 public:
151 using OutputCodeVariant = std::variant<HtmlCodeGenerator,
152 LatexCodeGenerator,
153 RTFCodeGenerator,
154 ManCodeGenerator,
155 XMLCodeGenerator,
156 DocbookCodeGenerator,
157 DevNullCodeGenerator,
158 OutputCodeDeferHtml,
159 OutputCodeDeferLatex,
160 OutputCodeDeferRTF,
161 OutputCodeDeferMan,
162 OutputCodeDeferDocbook,
163 OutputCodeDeferExtension
164 >;
165
166 int id() const { return m_id; }
167 void setId(int id) { m_id = id; }
168
169 /** Add a code generator already wrapped in a variant type. */
170 void add(OutputCodeVariant v)
171 {
172 m_outputCodeList.push_back(OutputCodeElem(v,true));
173 }
174
175 /** Add a code generator to the list, using a syntax similar to std::make_unique<T>() */
176 template<class T,class... As>
177 void add(As&&... args)
178 {
179 add(OutputCodeVariant{std::in_place_type<T>,std::forward<As>(args)...});
180 }
181
182 /** Returns a pointer to a specific generator in the list */
183 template<class T>
184 T *get()
185 {
186 for (auto &e : m_outputCodeList)
187 {
188 T *t = std::get_if<T>(&e.variant);
189 if (t) return t;
190 }
191 return nullptr;
192 }
193
194 /** Enable or disable a specific generator */
195 void setEnabledFiltered(OutputType o,bool enabled)
196 {
197 for (auto &e : m_outputCodeList)
198 {
199 std::visit([&](auto &&arg) { if (arg.type()==o) e.enabled = enabled; },e.variant);
200 }
201 }

242
243 private:
244 template<template <class> class GeneratorT, class... As>
245 void foreach(As&&... args)
246 {
247 for (auto &e : m_outputCodeList)
248 {
249 if (e.enabled)
250 {
251 dispatch_call<GeneratorT>(e.variant,std::forward<As>(args)...);
252 }
253 }
